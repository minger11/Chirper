#For websocket connections, a websocket handshake request is sent by the client to the server

#The first line of the request looks like a standard HTTP request, with method (GET), path (/ws?topic=) and protocol version (HTTP/1.1)
GET /ws?topic= HTTP/1.1

#A request header containing the server address as the client understands it (as the server might have multiple domains mapped to the same IP address)

#The user agent string of the user agent, generated by the browser. Identifies browser and provides certain system details to the server
User­Agent: Mozilla/5.0 (X11; Fedora; Linux x86_64; rv:41.0) Gecko/20100101 Firefox/41.0

#Specifies which media types are acceptable for the response
Accept: text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8

#Restricts the set of natural languages that are preferred as a response
Accept-Language: en­US,en;q=0.5

#Restricts the content-codings that are acceptable in response

#A required field to specify the version and MUST BE 13

#Sppecifies the desired options for this particular connection

#Specifies that the request SHOULD be forwarded by the application to the origin server even if it has a cached copy of the request
Pragma: no-cache

#A cache MUST NOT use the response to satisfy future requests without a revalidation from the server
Cache-Control: no-cache

#Clients specification of what communication protocols it supports and would like to use if the server switches protocols
